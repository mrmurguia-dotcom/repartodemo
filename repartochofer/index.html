function guardarNovedad(info) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName("CP " + info.cp); if (!sheet) return "Error"; // LISTA DE FILAS A ACTUALIZAR (si es multiple o solo una) const filasAProcesar = info.filasMultiple || [info.fila]; let urlFile = ""; if (info.archivoB64) { const folder = DriveApp.getFolderById(FOLDER_FOTOS_ID); const blob = Utilities.newBlob(Utilities.base64Decode(info.archivoB64.split(",")[1]), "image/jpeg", "NOVEDAD_" + info.factura); urlFile = folder.createFile(blob).setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW).getUrl(); } // Aplicamos los cambios a todas las filas seleccionadas filasAProcesar.forEach(filaNro => { sheet.getRange(filaNro, 8).setValue(info.status); // Status sheet.getRange(filaNro, 9).setValue(info.chofer); // Chofer sheet.getRange(filaNro, 10).setValue(info.motivo || ""); // Comentario if (urlFile) sheet.getRange(filaNro, 11).setValue(urlFile); // Foto }); return "OK"; }
